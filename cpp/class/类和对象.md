```cpp
#include <iostream>
class A{
public:
    virtual void check() {
        std::cout << "A" << std::endl;
    }
    void check_no_virtual() {
        std::cout << "A_no" << std::endl;
    }
};
class B: public A{
public:
    virtual void check() {
        std::cout << "B" << std::endl;
    }
    void check_no_virtual() {
        std::cout << "B_no" << std::endl;
    }
};
int main() {
    std::cout << "Hello, World!" << std::endl;
    B b;
    A a;
    A &a1 = a;
    A &a2 = b;
    a1.check(); // A; 跟实际对象有关
    a2.check(); // B
    std::cout <<  "===" << std::endl;
    a1.check_no_virtual(); // A_no; 跟引用变量的类型有关
    a2.check_no_virtual(); // A_no
    return 0;
}
```
